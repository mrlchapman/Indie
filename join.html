function dropWord(event) { event.preventDefault(); const word = event.dataTransfer.getData('text/plain'); // Check if dropping into the answer area or onto another word tile if (event.target.id === 'answer-area' || event.target.classList.contains('word-tile')) { let wordTile; // If dragging from scrambled words if (draggedElement.parentElement.id === 'scrambled-words') { // Create a new word tile in the answer area wordTile = document.createElement('div'); wordTile.classList.add('word-tile'); wordTile.textContent = word; wordTile.draggable = true; wordTile.addEventListener('dragstart', dragStart); wordTile.addEventListener('dragover', dragOver); wordTile.addEventListener('drop', dropWord); wordTile.addEventListener('touchstart', touchStart); wordTile.addEventListener('touchmove', touchMove); wordTile.addEventListener('touchend', touchEndAnswer); wordTile.addEventListener('dblclick', removeWord); // Remove the word from scrambled words draggedElement.remove(); } else { // If dragging within answer area wordTile = draggedElement; } const answerArea = document.getElementById('answer-area'); if (event.target.classList.contains('word-tile') && event.target.parentElement.id === 'answer-area') { // Insert before or after the target word tile based on position const targetTile = event.target; const draggedIndex = Array.from(answerArea.children).indexOf(wordTile); const targetIndex = Array.from(answerArea.children).indexOf(targetTile); if (draggedIndex < targetIndex) { answerArea.insertBefore(wordTile, targetTile.nextSibling); } else { answerArea.insertBefore(wordTile, targetTile); } } else { // Append to the end if dropped in empty area answerArea.appendChild(wordTile); } } }
